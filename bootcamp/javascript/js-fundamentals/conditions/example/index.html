Operators

Equal

        
            const a = 3;
            const b = '3';

            console.log(a == b);
            // true

Not equal

        
            const a = 1;
            const b = 2;

            console.log(a != b);
            // true
        
    
        
            const c = 3;
            const d = 3;

            console.log(a != b);
            // false

Strict equal

        
            const a = 3;
            const b = '3';

            console.log(a === b);
            // false
        
    
        
            const a = 3;
            const b = 3;

            console.log(a === b);
            // true

Strict not equal

        
            const a = 3;
            const b = 3;

            console.log(a !== b);
            // false
        
    
        
            const a = 3;
            const b = '3';

            console.log(a !== b);
            // true

Greater than (>)

        
            const a = 3;
            const b = 1;

            console.log(a > b);
            // true

Greater than or equal (>=)

        
            const a = 2;
            const b = 2;

            console.log(a >= b);
            // true

Less than (<)

        
            const a = 1;
            const b = 2;

            console.log(a < b);
            // true

Less than or equal (<=)

        
            const a = 3;
            const b = 1;

            console.log(a <= b);
            // false

Truthy or Falsy

Truthy

        
            console.log(true);
            console.log(!! []);
            console.log(!! {});
            console.log(!! 42);
            console.log(!! "0");
            console.log(!! "false");
            console.log(!! new Date());
            console.log(!! -42);
            console.log(!! 12n);
            console.log(!! 3.14);
            console.log(!! -3.14);
            console.log(!! Infinity);
            console.log(!! -Infinity);

Falsy

        
            console.log(false);
            console.log(!! null);
            console.log(!! undefined);
            console.log(!! 0);
            console.log(!! -0);
            console.log(!! 0n);
            console.log(!! NaN);
            console.log(!! "");

Logical Operators

AND

        
            if (choice === 'sunny' && temperature < 86) {
                para.textContent = 'It is ' + temperature + ' degrees outside — nice and sunny. Let\'s go out to the beach, or the park, and get an ice cream.';
            } else if (choice === 'sunny' && temperature >= 86) {
                para.textContent = 'It is ' + temperature + ' degrees outside — REALLY HOT! If you want to go outside, make sure to put some sunscreen on.';

            if (iceCreamVanOutside || houseStatus === 'on fire') {
