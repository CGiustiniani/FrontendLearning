## Equal
```js
const a = 3;
const b = '3';

console.log(a == b);
// true
```

## Not equal
```js
const a = 1;
const b = 2;

console.log(a != b);
// true
```

```js
const c = 3;
const d = 3;

console.log(a != b);
// false
```

## Strict equal
```js
const a = 3;
const b = '3';

console.log(a === b);
// false
```

```js
const a = 3;
const b = 3;

console.log(a === b);
// true
```

## Strict not equal
```js
const a = 3;
const b = 3;

console.log(a !== b);
// false
```

```js
const a = 3;
const b = '3';

console.log(a !== b);
// true
```

## Greater than (>)
```js
const a = 3;
const b = 1;

console.log(a > b);
// true
```

## Greater than or equal (>=)
```js
const a = 2;
const b = 2;

console.log(a >= b);
// true
```

## Less than (<)
```js
const a = 1;
const b = 2;

console.log(a < b);
// true
```

## Less than or equal (<=)
```js
const a = 3;
const b = 1;

console.log(a <= b);
// false
```
