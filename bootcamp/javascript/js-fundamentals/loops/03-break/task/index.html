The following example iterates through the elements in an array until it finds the index of an element whose value is theValue:

        
            const arr = [23, 56, 5, 87, 245];
            const theValue = 3;

            for (let i = 0; i < arr.length; i++) {
                if (arr[i] === theValue) {
                    break;
                }
            }
